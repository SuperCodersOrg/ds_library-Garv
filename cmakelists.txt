cmake_minimum_required(VERSION 3.10)

project(DSLibrary)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

enable_testing()

add_compile_definitions(GTEST_HAS_PTHREAD=0)

include(FetchContent)
FetchContent_Declare(
    googletest
    GIT_REPOSITORY https://github.com/google/googletest.git
    GIT_TAG v1.17.0
)

set(gtest_force_shared_crt ON CACHE BOOL "" FORCE) 
set(gtest_disable_pthreads ON CACHE BOOL "" FORCE) 
FetchContent_MakeAvailable(googletest)

#3.TEST DA
add_executable(datest
    tests/dynamicarray/test_append.cpp
    tests/dynamicarray/test_constructors.cpp
    tests/dynamicarray/test_get_set.cpp
    tests/dynamicarray/test_remove.cpp
    tests/dynamicarray/test_insert.cpp
    tests/dynamicarray/test_iterators.cpp
    tests/dynamicarray/test_operator.cpp
    tests/dynamicarray/test_utility.cpp
)

target_include_directories(datest PRIVATE include)

target_link_libraries(datest gtest_main)

include(GoogleTest)
gtest_discover_tests(datest)

#4.TEST SL
add_executable(sltest
